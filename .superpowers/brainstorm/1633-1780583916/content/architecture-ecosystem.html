集成中心核心模型

# integration/models/

class ConnectorDefinition(CoreModel):           # 连接器定义
    code     = CharField(unique=True)          # aliyun_ecs / wecom / sms_aliyun
    name     = CharField()                     # 阿里云ECS / 企业微信
    category = CharField(choices=CATEGORY)      # cloud / notification / auth / paas
    version   = CharField()
    icon      = CharField()                     # 图标 URL
    config_schema = JSONField()               # 配置表单 JSON Schema
    is_active = BooleanField(default=True)
    provider_class= CharField()                # Python 类路径

class ConnectorInstance(CoreModel):             # 连接器实例 (一个定义可多个实例)
    definition = FK(ConnectorDefinition)
    name       = CharField()                   # "生产环境阿里云"
    config     = JSONField()                   # region, endpoint 等非敏感配置
    status     = CharField(choices=STATUS)      # online / offline / error
    health_check_url = CharField()
    last_health_check= DateTimeField()
    is_active  = BooleanField(default=True)

class ConnectorCredential(CoreModel):          # 凭证 (加密存储)
    instance   = FK(ConnectorInstance)
    name       = CharField()                   # "阿里云 AK"
    cred_type  = CharField(choices=CRED_TYPE)   # access_key / password / token / cert
    encrypted_value = TextField()            # AES-256 加密存储
    expire_at  = DateTimeField(null=True)     # 过期时间

class IntegrationLog(CoreModel):               # 调用审计日志
    instance   = FK(ConnectorInstance)
    action     = CharField()                   # describe_instances / send_sms
    request    = JSONField()
    response   = JSONField()
    status     = CharField(choices=CALL_STATUS) # success / failed / timeout
    duration_ms= IntegerField()
    error_msg  = TextField(null=True)
    source     = CharField()                   # 调用来源模块

📐 集成中心的设计原则

  • 所有外部调用走集成中心 — 不准其他模块直接调用外部 API，强制通过集成中心
  • 凭证统一加密管理 — AES-256 加密存储，审计所有凭证使用记录
  • 连接器可插拔 — 每个适配器是一个 Python class，注册后自动发现
  • 统一熔断/限流 — 防止外部 API 不稳定级联影响整个平台
  • 健康检查 — 定期检测外部服务可达性，状态可视化
