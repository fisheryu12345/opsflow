neomodel 节点定义 (示意)

# cmdb/models/node_types.py

class Biz(StructuredNode):
    biz_id       = UniqueIdProperty()    # 业务ID
    name         = StringProperty()       # 业务名称
    lifecycle    = StringProperty(choices=LIFECYCLE)  # 生命周期
    operator     = StringProperty()
    description  = StringProperty()
    created_at   = DateTimeProperty(default_now=True)

    # 关系
    sets = RelationshipTo('Set', 'CONTAINS')

class Host(StructuredNode):
    host_id      = UniqueIdProperty()
    ip           = StringProperty(required=True)
    hostname     = StringProperty()
    os_type      = StringProperty()
    cpu_cores    = IntegerProperty()
    memory_mb    = IntegerProperty()
    disk_gb      = IntegerProperty()
    status       = StringProperty(choices=HOST_STATUS)
    agent_status = StringProperty(choices=AGENT_STATUS)
    agent_version= StringProperty()
    labels       = JSONProperty()       # 自定义标签

    # 关系
    module  = RelationshipTo('Module', 'BELONGS_TO')
    process = RelationshipTo('Process', 'RUNS')
🛈 所有节点继承 StructuredNode，关系通过 RelationshipTo/From 定义
