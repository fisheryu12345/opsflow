三、X6 实现方案

// Custom markup: 6 layers of SVG elements
const markup = [
  { tagName: 'rect', selector: 'shadow' },       // drop shadow (offset rect)
  { tagName: 'rect', selector: 'accent-bar' },    // left colored bar (4px)
  { tagName: 'rect', selector: 'body' },           // main card body
  { tagName: 'circle', selector: 'icon-bg' },      // icon background circle
  { tagName: 'text', selector: 'icon' },            // icon emoji/SVG
  { tagName: 'text', selector: 'label' },            // node title
  { tagName: 'text', selector: 'subtitle' },         // plugin group + risk
  { tagName: 'circle', selector: 'status-dot' },     // status indicator dot
]

// Node.define with attrs for each selector
const NodeCard = Shape.Rect.define({
  shape: 'ops-atom',
  width: 220,
  height: 60,
  markup,
  attrs: {
    body: { fill: '#FFF', stroke: '#E4E7ED', ... },
    'accent-bar': { fill: '#409EFF', ... },
    'icon-bg': { fill: '#ECF5FF', stroke: '#D9ECFF', ... },
    label: { text: 'Node', ... },
    subtitle: { text: 'group · risk', ... },
    'status-dot': { r: 4, fill: 'transparent' },  // hidden in design mode
  },
})

// Monitor mode: just update attrs based on execution status
node.setAttrs({
  body: { fill: '#F0F9EB', stroke: '#67C23A' },
  'accent-bar': { fill: '#67C23A' },
  label: { fill: '#67C23A' },
  subtitle: { fill: '#67C23A', text: '已完成 · 耗时 1.8s' },
  'status-dot': { fill: '#67C23A' },
})

// Running animation: X6 + CSS keyframes
node.setAttrs({
  'icon-bg': { stroke: '#E6A23C', strokeDasharray: '60 28' }
}) // with animateTransform for rotation

五、运行态 vs 设计态的差异总结

🎨 设计态

  • 白色卡片 + 浅灰边框
  • 左色条显示插件分组颜色
  • 图标 + 标题 + 副标题（分组名）
  • Port 默认半透明，hover 展开
  • 风险等级小圆点右上角

▶️ 运行态（在以上基础上叠加）

  • 边框颜色 → 执行状态色
  • 背景填充 → 状态浅色背景
  • 左色条 → 状态色
  • 文字颜色 → 状态色
  • 副标题 → 状态描述（执行中/耗时/错误）
  • 图标区域 → 运行中旋转动画
  • 右上角状态圆点 pulse 动画
  • Port 在运行态隐藏（不可编辑）
